function Clock() {
  const [time, setTime] = useState(new Date());

  useEffect(() => {
    // Effect: ‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô‡∏Å‡∏≤‡∏£‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏Ç‡∏≠‡∏á interval

    // ‚≠êÔ∏è Cleanup Function ‚≠êÔ∏è
    // ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏ô‡∏µ‡πâ‡∏à‡∏∞‡∏ñ‡∏π‡∏Å‡πÄ‡∏£‡∏µ‡∏¢‡∏Å‡πÄ‡∏°‡∏∑‡πà‡∏≠ Component ‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏à‡∏∞ unmount

      console.log('Clock component unmounted. Timer stopped.');

  }, []); // [] -> ‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏Ñ‡∏£‡∏±‡πâ‡∏á‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏ï‡∏≠‡∏ô mount ‡πÅ‡∏•‡∏∞ cleanup ‡∏ï‡∏≠‡∏ô unmount

  return (
    <div>
